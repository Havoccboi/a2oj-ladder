vector <string> s;
vector <string> q;

	int n;
	cin>>n;

	for(int i = 0 ;i< n ; i++){
	
	cin>>tt;
	if(tt[0] != '#'){
		 f = true;

	}

	}

		cout<<s.at(i)<<" ";
